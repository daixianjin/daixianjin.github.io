摘要

在一个项目当中，列表数据的搜索往往是比较常见的一种功能，也是一个难处理的功能，反正我觉得特别难处理。一个列表搜索可能就会写很多，也可以越加越多，特别难复用之后代码越来越多的冗余。我在的处理比较笨，那就在一个服务层里面写上所有可能用到的搜索条件，以下是代码的展示（请勿嘲笑）：

<?php  

        public function searchService(object $query)
    {
        $page  = request('page', 1);
        $limit = request('limit', 20);
        $query = $query::query();

        if ($mobile = request('mobile'))
        {
            $query->where([
                'mobile' => $mobile
            ]);
        }

        if ($inviteMobile = request('invite_mobile'))
        {
            $query->where([
                'id' => User::where([
                    'mobile' => $inviteMobile

        }

        if ($address = request('address'))
        {
            $query->where([
                'address' => $address
            ]);
        }

        if ($erhAddress = request('eth_address'))

            $query->where([
                'eth_address' => $erhAddress
            ]);
        }

        if ($googleKey = request('google_key'))
        {
            $query->where([
                'google_key' => $googleKey
            ]);
        }

        if (in_array($googleStatus = request('google_status',3),[0,1]))
        {
            $query->where([
                'google_status' => $googleStatus
            ]);
        }

        if ($nickname = request('nickname'))
        {
            $query->where([
                'nickname' => $nickname
            ]);
        }

        if ($identity = request('user_identity'))
        {
            $query->where([
                'user_identity' => $identity
            ]);
        }

        if ($inviteCode = request('invite_code'))
        {
            $query->where([
                'invite_code' => $inviteCode
            ]);

        if (in_array($status = request('status',3),[0,1]))

        if($time = request('created_at'))

Eloquentfilter

Eloquentfilter 一种有说服力的过滤 Laravel 模型及其关系的方法的扩展包，能很好的帮助我们处理好条件搜索的各种问题，而且也可能很好做到的复用。

安装

composer require tucker-eric/eloquentfilter

在 config/app.php 配置文件中注册服务提供者:

'providers' => [
    EloquentFilter\ServiceProvider::class,
],

使用 publish 命令将包配置文件复制到本地配置文件:

php artisan vendor:publish --provider="EloquentFilter\ServiceProvider"

发布配置文件后，会在 config 目录当中看见生成的配置文件 config/eloquentfilter.php

# 设置模型的命令空间，如果要改到自己模型目录当中可以这样

'namespace' => "App\\ModelFilters\\",

# 修改到 Models 目录下

'namespace' => "App\\Models\Filters\\",

生成

扩展已经提供了生成模板的命令，执行命令后会自动生成对应的 Filter。

php artisan model:filter UserFilter

所有过滤器的默认名称空间是 App ModelFilters，每个 Model 都希望过滤器类名遵循 { $ModelName } 过滤器变数命名原则，而不管模型在哪个名称空间。下面是一个基于默认变数命名原则的模型和它们各自的过滤器的例子。

Model ModelFilter
App\User App\ModelFilters\UserFilter
App\FrontEnd\PrivatePost App\ModelFilters\PrivatePostFilter
App\FrontEnd\Public\GuestPost App\ModelFilters\GuestPostFilter

使用

已经生成好了 Filter 之后改如何去使用呢？

<?php

namespace App\Model;

use EloquentFilter\Filterable;
use Illuminate\Database\Eloquent\Model;

class UserRole extends Model
{
    use Filterable; # use 扩展提供的 trait

    protected $guarded = [];

    # 创建一个公共的方法并引入创建的 Filter
    public function modelFilter()
    {
        return $this->provideFilter(\App\Model\Filters\UserFilter::class);
    }
}

以下是定义模型使用 Filter ，创建 Filter 肯定是要拿来过滤搜索。

<?php

namespace App\Model\Filters;

use EloquentFilter\ModelFilter;

class UserFilter extends ModelFilter
{
    use CommonFilter; # 使用一个 trait 定义一些公共的条件
    /**
    * Related Models that have ModelFilters as well as the method on the ModelFilter
    * As [relationMethod => [input_key1, input_key2]].
    *
    * @var array
    */
    public $relations = [''];
        protected $blacklist = ['name']; # 黑名单，禁止使用的方法 $this->blacklistMethod() 扩展包提供的添加黑名单，$this->whitelistMethod() 扩展包提供的删除黑名单。嘿嘿，白色，黑色 政治不正确哦，哈哈哈哈哈哈哈哈哈！

        # 看着这里注意：！！方法名 ！！就是前段传递的搜索的字段
    public function name($name)
    {
        return $this->whereLike('role_name',$name); 
    }

      # 看着这里注意：！！方法名 ！！就是前段传递的搜索的字段
    public function introduction($introduction)
    {
        return $this->whereLike('introduction',$introduction);
    }

    # 看着这里注意：！！方法名 ！！就是前段传递的搜索的字段
    public function role($role)
    {
        # 如果字段的以 _id 结尾可以省略 _id ,如果你不想这样可以 protected $drop_id = false;
        return $this->whereLike('role_id',$role);
    }

    # 看着这里注意：setup 很特殊，只要定义了 setup 那么他都会每次使用这个 filter 的时候都会去执行一次
    public function setup()
    {
        return $this->with('roles')
    }

    # $this->input('x') # 获取前端传递的某个字段
    # $this->push('x','xx') # 如果在 setup 方法中调用，则添加要过滤的值。如果调用其他方法，则只会过滤关系
}
<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use App\Http\Requests;
use App\User;
use App\ModelFilters\Admin\UserFilter as AdminFilter;
use App\ModelFilters\User\UserFilter as BasicUserFilter;
use Auth;

class UserController extends Controller
{      
      # 使用方法1
      public function test1(Request $request)
    {
                User::filter($request->all())->get();
    }

      # 使用方法2，可以通过传递要用作 filter() 方法的第二个参数的筛选器来动态定义筛选器。动态定义过滤器将比为模型定义的任何其他过滤器采用先例。  
    public function test2(Request $request)
    {
        $userFilter = Auth::user()->isAdmin() ? AdminFilter::class : BasicUserFilter::class;

        return User::filter($request->all(), $userFilter)->get();
    }
}

这时候 filter 提供的模型关联过滤就很好用了，请看以下操作：

# 用户模型，我们关联角色，角色模型可以不做任何操作
class User extends Model
{
      use Filterable;

        public function roles()
    {
        $this->hasOne(Role::class);
    }
}

# 角色的 Filter ，我们只查询角色名称
class UserRoleFilter extends ModelFilter
{
    /**
    * Related Models that have ModelFilters as well as the method on the ModelFilter
    * As [relationMethod => [input_key1, input_key2]].
    *
    * @var array
    */
    public $relations = [];

    public function name($name)
    {
        return $this->whereLike('role_name',$name);
    }
}

# 用户 Filter，可以试用 $this->related(),但是我个人还是喜欢使用 $relations 数组
class UserFilter extends ModelFilter
{
      # roles 也就是用户模型当中模型关联，name 也就是 角色 Filter 当中的搜索字段
    public $relations = ['roles' => 'name'];
}
