Golang-字符串函数详解

Golang

更新日期:   2020-11-12

基本了解

跟 PHP 一样，Go 也为我们准备了很多系统函数，把我们常使用的功能都已经给我封装好了，非常的利于我们开发。下面介绍的都是属于开发中常用的字符串函数。

len - 字符串长度

统计字符串长度和数组长度，按字节，注意 Go 的编码统一为 utf-8 ascii 码的字符（字母和数字）占一个字节，汉字占用 3 个字节。

func main() {
    arr := [4]int{1, 2, 3}
    str := "123你"
    fmt.Printf("长度为多少个字节：%v\n", len(str))
    fmt.Printf("长度为多少：%v\n", len(arr))
}

// 长度为多少个字节：6
// 长度为多少：4

[]rune - 遍历字符串

字符串遍历，同时处理有中文的问题，其实也就是一个切片。

func main() {
    str := "9527周星驰"
    r := []rune(str)
    for i := 0; i < len(r); i++ {
        fmt.Printf("输出字符串:%c\n", r[i])
    }
}
// 输出字符串:9
// 输出字符串:5
// 输出字符串:2
// 输出字符串:7
// 输出字符串:周
// 输出字符串:星
// 输出字符串:驰

strconv.Atoi - 字符串转整数

func main() {
    n, err := strconv.Atoi("123")

    if err != nil {
        fmt.Printf("转换错误：%v", err)
    } else {
        fmt.Printf("转换成功：%v,类型是：%T", n, n)
    }
}

// 转换成功：123,类型是：int

strconv.Itoa - 整数转字符串

func main() {
    n := strconv.Itoa(123)
    fmt.Printf("转换成功：%v,类型是：%T", n, n)
}

// 转换成功：123,类型是：string

[]byte - 字符串转 ascii

func main() {
    n := []byte("123 ")
    fmt.Printf("转换成功：%v", n)
}

// 转换成功：[49 50 51 32]

[]byte - ascii 转字符串。

func main() {
    n := string([]byte{78, 77, 76, 75})
    fmt.Printf("转换成功：%v", n)
}

// 转换成功：NMLK

strconv.ForamInt - 10 进制转 2，8，16 进制

func main() {
    n := strconv.FormatInt(1, 2)
    fmt.Printf("1 转换成 2 进制 = %v\n", n)
    n1 := strconv.FormatInt(1, 8)
    fmt.Printf("1 转换成 8 进制 = %v\n", n1)
    n2 := strconv.FormatInt(1, 16)
    fmt.Printf("1 转换成 16 进制 = %v\n", n2)
}

// 1 转换成 2 进制 = 1
// 1 转换成 8 进制 = 1
// 1 转换成 16 进制 = 1

strings.Contains - 字符串查找

func main() {
    str := "你好世界呀，zhouxingchi"
    b := strings.Contains(str, "zhou")
    fmt.Printf("是否查找到字符串：%v", b)
}

// 是否查找到字符串：true

strings.Count - 查找字符串出现的次数

func main() {
    str := "你好世界呀，zhouxingchi"
    b := strings.Count(str, "i")
    fmt.Printf("字符串出现的次数：%v", b)
}

// 字符串出现的次数：2

srings.EquelFold - 字符串比较（不区分大小写）

func main() {
    str := "abc"
    b := strings.EqualFold(str, "ABC")
    fmt.Printf("它们是否相等：%v", b)
}

// 它们是否相等：true

strings.Index - 查询字符串第一次出现的下标

func main() {
    str := "你好世界呀，zhouxingchi"
    b := strings.Index(str, "世")
    fmt.Printf("字符串下标：%v", b)
}

// 字符串下标：6

strings.LastIndex - 查询字符串最后出现的下标

func main() {
    str := "golang"
    b := strings.LastIndex(str, "g")
    fmt.Printf("字符串下标：%v", b)
}

// 字符串下标：5

strings.Replace - 字符串替换

func main() {
    str := "go golang"
    b := strings.Replace(str, "go", "php", 1) // -1 表示全部替换
    fmt.Printf("替换后的结果：%v", b)
}

// 替换后的结果：php golang

strings.Split - 字符串转数组

func main() {
    str := "go golang"
    b := strings.Split(str, " ")
    fmt.Printf("转换后的结果：%v", b)
}

// 转换后的结果：[go golang]

strings.Tolower - 字符串转小写

func main() {
    str := "Go golang"
    b := strings.ToLower(str)
    fmt.Printf("替换后的结果：%v", b)
}

// 转换后的结果：go golang

strings.Upper - 字符串转大写

func main() {
    str := "Go golang"
    b := strings.ToUpper(str)
    fmt.Printf("转换后的结果：%v", b)
}

// 转换后的结果：GO GOLANG

strings.TrimSpace - 去除首尾空格

func main() {
    str := " golang "
    b := strings.TrimSpace(str)
    fmt.Printf("去除后的结果:%v", b)
}

// 去除后的结果:golang

strings.Trim - 去除首尾指定字符

func main() {
    str := "golang"
    b := strings.Trim(str, "g")
    fmt.Printf("去除后的结果:%v", b)
}

// 去除后的结果:olan

strings.TrimLeft - 去除首指定字符

func main() {
    str := "golang"
    b := strings.TrimLeft(str, "g")
    fmt.Printf("去除后的结果:%v", b)
}

// 去除后的结果:olang

strings.TrimRight - 去除尾指定字符

func main() {
    str := "golang"
    b := strings.TrimRight(str, "g")
    fmt.Printf("去除后的结果:%v", b)
}

// 去除后的结果:golan

strings.HasSuffix - 判断字符是否是指定字符串结尾

func main() {
    str := "golang.jpg"
    b := strings.HasSuffix(str, ".jpg")
    fmt.Printf("是否是指定字符串结尾:%v", b)
}

// 是否是指定字符串结尾:true

strings.HasPrefix - 判断字符是否是指定字符串开头

func main() {
    str := "golang.jpg"
    b := strings.HasPrefix(str, "g")
    fmt.Printf("是否是指定字符串开头:%v", b)
}

// 是否是指定字符串开头:true
