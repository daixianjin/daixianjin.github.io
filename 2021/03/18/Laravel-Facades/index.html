摘要

设计模式，是一套建立在长期对代码设计的实践基础上，能够在众多应用场景下反复套用的经验总结。 可以说，设计模式能够保证代码的健壮性，提高程序的可扩展性，是程序设计里的最佳模型和优秀解决方案。 设计模式比之于高楼大厦，就是其中的钢筋混泥土结构，是代码工程化的基石。

仓库模式

PHP 使用仓库模式好嘛，为什么要使用仓库模式呢？

答：我个人认为这是一个伪命题，在以现在的 PHP 框架来说（MVC）模式已经很简单了，为什么还需要加这么一层模式呢？ORM 已经非常简洁，PHP 的最大特点就是高效率的开发，为什么还要加一层仓库模式呢？为什么还需要这么多度去封装呢？为什么不享受这简洁高效的开发呢？。这个问题，问 10 个 PHP 开发者，绝对没有统一的答案，所以我说这个是伪命题。

但是我个人还是偏向与项目当中使用仓库模式，仓库模式使数据访问逻辑分离并将其映射到业务逻辑中的业务实体。数据访问逻辑和业务逻辑之间的通信通过接口来完成。

仓库模式 Repository Pattern 主要思想是建立一个数据操作代理层，把 Service 里的数据操作剥离出来，这样做有几个好处：

  • 把数据处理逻辑分离使得代码更容易维护
  • 数据处理逻辑和业务逻辑分离
  • 减少代码重复
  • 降低代码出错的几率
  • 让 Service 代码的可读性大大提高

在 Contracts 当中创建仓库模式的接口实现

<?php

namespace App\Repositories\Contracts;

interface RepositoryInterface
{
    /**
     * 查询一条数据总数量
     * @param array $data
     * @param bool $or
     * @param string $columns
     *
     * @return int
     * */
    public function count(array $data,bool $or,string $columns) : int;


    /**
     * 添加数据，可返回模型
     * @param array $data
     *
     * @return object
     */
    public function create(array $data) : object;

    /**
     * 添加数据，不返回模型
     * @param array $data
     *
     * @return void
     * */
    public function insert(array $data) : void;

    /**
     * 根据主键，真删除
     * @param array $data
     * @param bool $or
     *
     * @return mixed
     */
    public function delete(array $data,bool $or = false);

    /**
     * 查询数据过滤并且分页
     * @param array $data
     * @param array $columns
     * @param bool $toArray
     *
     * @return mixed
     * */
    public function filter(array $data,array $columns,bool $toArray = true);

    /**
     * 无需模型更新
     * @param array $data
     * @param array $columns
     * @param bool $or
     *
     * @return int
     */
    public function update(array $data, array $columns,bool $or) : int;

    /**
     * 存在更新不存在插入
     * @param array $data
     * @param array $columns
     *
     * @return void
     * */
    public function updateOrInsert(array $data,array $columns) : void;

    /**
     * 无需模型更新(in)
     * @param array $data
     * @param array $in
     * @param array $update
     * @param bool $or
     *
     * @return int
     * */
    public function updateByWhereIn(array $data,array $in,array $update,bool $or = false) : int;

    /**
     * 根据主键查询一条数据
     * @param int $id
     * @param array $columns
     *
     * @return mixed
     */
    public function find(int $id, array $columns = ['*']);

    /**
     * 根据条件查询一条数据
     *
     * @param array $data
     * @param array $columns
     * @param bool $or
     *
     * @return mixed
     */
    public function findByWhere(array $data, array $columns = ['*'], $or = false);

    /**
     * 根据条件查询一条数据(in)
     * @param array $data
     * @param array $in
     * @param array $columns
     * @param bool $or
     *
     * @return mixed
     * */

    /**
     * 根据条件查询所有数据
     * @param array $data
     * @param array $columns
     * @param bool $or
     *
     * @return object
     */
    public function findByWhereAll(array $data, array $columns = ['*'],bool $or = false) : object;

    /**
     * 根据条件查询所有数据(in)
     * @param array $data
     * @param array $in
     * @param array $columns
     * @param bool $or
     *
     * @return object
     * */
    public function findByWhereInAll(array $data,array $in,array $columns = ['*'],bool $or = false) : object;
}

在 Eloquent 当中实现 Contracts 当中的接口的方法

<?php

namespace App\Repositories\Eloquent;

use Illuminate\Support\Collection;
use Illuminate\Container\Container;
use Illuminate\Database\Eloquent\Model;
use App\Repositories\Contracts\RepositoryInterface;

abstract class BaseRepository implements RepositoryInterface
{
    /**
     * 容器
     * @var object
     * */
    protected object $app;

    /**
     * 模型
     * @var object
     * */
    protected object $model;

    /**
     * 实例容器
     *
     * @throws \Exception
     */
    public function __construct()
    {
        $this->app = new Container();
        $this->criteria = new Collection();
        $this->makeModel();
    }

    /**
     * 模型实现的抽象方法
     *
     * @return string
     * */
    abstract function model() : string;

    /**
     * 获取模型
     *
     * @return object
     * */
    protected function getModel() : object
    {
        return $this->model;
    }

    /**
     * 重置模型
     *
     * @return void
     * @throws \Exception
     */
    protected function resetModel() : void
    {
        $this->makeModel();
    }

    /**
     * 容器的管理
     *
     * @throws \Exception
     * @return Model
     */
    protected function makeModel() : Model
    {
        $model = $this->app->make($this->model());

        if (! $model instanceof Model)
        {
            global_exception('Instance container error');
        }

       return $this->model = $model;
    }

    /**
     * 查询所有（慎重）- 要删除
     * @param array $columns
     *
     * @return object
     * @throws \Exception
     */
    public function all(array $columns = ['*']) : object
    {
        $res = $this->model->get($columns);

        $this->resetModel();

        return $res;
    }

    /**
     * 模型关联
     *
     * @param array $relations
     * @return $this
     */
    public function with(array $relations) : BaseRepository
    {
        $this->model = $this->model->with($relations);

        return $this;
    }

    /**
     * 加锁 (true 为悲观锁，false 为乐观锁)
     * @param bool $lock
     *
     * @return $this
     * */
    public function lock(bool $lock = true): BaseRepository
    {
        $this->model = $this->model->lock($lock);

        return $this;
    }

    /**
     * 查询一条数据总数量
     * @param array $data
     * @param bool $or
     * @param string $columns
     *
     * @return int
     * @throws \Exception
     */
    public function count(array $data,bool $or = false,string $columns = '*') : int
    {
        $this->applyWhereConditions($data,$or);

        $res = $this->model->count($columns);
        $this->resetModel();

        return $res;
    }

    /**
     * 添加数据，可返回模型
     * @param array $data
     *
     * @return object
     * @throws \Exception
     */
    public function create(array $data) : object
    {
        $res = $this->model->create($data);

        $this->resetModel();

        return $res;
    }

    /**
     * 添加数据，不返回模型
     * @param array $data
     *
     * @return void
     * @throws \Exception
     */
    public function insert(array $data) : void
    {
        $this->model->insert($data);

        $this->resetModel();
    }

    /**
     * 根据主键，真删除
     * @param array $data
     * @param bool $or
     *
     * @return mixed
     * @throws \Exception
     */
    public function delete(array $data,bool $or = false)
    {
        $this->applyWhereConditions($data,$or);

        $res = $this->model->delete();
        $this->resetModel();

        return $res;
    }

    /**
     * 查询数据过滤并且分页
     *
     * @param array $data
     * @param array $columns
     * @param bool $toArray
     * @return mixed
     * @throws \Exception
     */
    public function filter(array $data,array $columns,bool $toArray = true)
    {
        $limit = request('limit', 15);

        $res = $this->model->filter(request()->only($data))
            ->paginate($limit > 15 ? 15 : $limit,$columns , 'page', request('page', 1));

        if ($toArray) {
            $res = $res->toArray();
        }

        $this->resetModel();

        return $res;
    }

    /**
     * 无需模型更新
     * @param array $data
     * @param array $columns
     * @param bool $or
     *
     * @return int
     * @throws \Exception
     */
    public function update(array $data,array $columns,bool $or = false) : int
    {
        $this->applyWhereConditions($data,$or);
        $res = $this->model->update($columns);

        $this->resetModel();
        return $res;
    }

    /**
     * 存在更新不存在插入
     * @param array $data
     * @param array $columns
     *
     * @return void
     * @throws \Exception
     */
    public function updateOrInsert(array $data,array $columns) : void
    {
        $this->model->updateOrInsert($data,$columns);

        $this->resetModel();
    }

    /**
     * 按ID批量更新
     * @param array $data
     * @param array $in
     * @param array $update
     * @param bool $or
     *
     * @return int
     * @throws \Exception
     */
    public function updateByWhereIn(array $data,array $in,array $update,bool $or = false) : int
    {
        $this->applyWhereConditions($data,$or);

        $this->applyWhereInConditions($in);

        $res = $this->model->update($update);

        $this->resetModel();

        return $res;
    }

    /**
     * 根据主键查询一条数据
     * @param int $id
     * @param array $columns
     *
     * @return mixed
     * @throws \Exception
     */
    public function find(int $id, array $columns = ['*'])
    {
        $res = $this->model->find($id, $columns);

        $this->resetModel();

        return $res;
    }

    /**
     * 根据条件查询一条数据
     *
     * @param array $data
     * @param array $columns
     * @param bool $or
     *
     * @return mixed
     * @throws \Exception
     */
    public function findByWhere(array $data, array $columns = ['*'], $or = false)
    {
        $this->applyWhereConditions($data,$or);

        $res = $this->model->first($columns);
        $this->resetModel();

        return $res;
    }

    /**
     * 根据条件查询一条数据(in)
     * @param array $data
     * @param array $in
     * @param array $columns
     * @param bool $or
     *
     * @return mixed
     * @throws \Exception
     */
    public function findByWhereIn(array $data,array $in,array $columns = ['*'],bool $or = false)
    {
        $this->applyWhereConditions($data,$or);

        $this->applyWhereInConditions($in);

        $res = $this->model->first($columns);

        $this->resetModel();

        return $res;
    }


    /**
     * 根据条件查询集合
     *
     * @param array $data
     * @param array $columns
     * @param bool $or
     *
     * @return object
     * @throws \Exception
     */
    public function findByWhereAll(array $data, array $columns = ['*'], $or = false) : object
    {
        $this->applyWhereConditions($data,$or);

        $res = $this->model->get($columns);
        $this->resetModel();

        return $res;
    }

    /**
     * 根据条件查询所有数据(in)
     * @param array $data
     * @param array $in
     * @param array $columns
     * @param bool $or
     *
     * @return object
     * @throws \Exception
     */
    public function findByWhereInAll(array $data,array $in,array $columns = ['*'],bool $or = false) : object
    {
        $this->applyWhereConditions($data,$or);

        $this->applyWhereInConditions($in);

        $res = $this->model->get($columns);

        $this->resetModel();

        return $res;
    }

    /**
     * 将给定的 where 条件应用于模型。
     * @param array $where
     * @param bool $or
     * @return void
     * @throws \Exception
     */
    protected function applyWhereConditions(array $where, bool $or = false) : void
    {
        foreach ($where as $field => $value) {
            if ($value instanceof \Closure) {
                $this->model = (!$or)
                    ? $this->model->where($value)
                    : $this->model->orWhere($value);
            } elseif (is_array($value)) {
                if (count($value) === 3) {
                    list($field, $operator, $search) = $value;
                    $this->model = (!$or)
                        ? $this->model->where($field, $operator, $search)
                        : $this->model->orWhere($field, $operator, $search);
                } elseif (count($value) === 2) {
                    list($field, $search) = $value;
                    $this->model = (!$or)
                        ? $this->model->where($field, '=', $search)
                        : $this->model->orWhere($field, '=', $search);
                }
            } else {
                $this->model = (!$or)
                    ? $this->model->where($field, '=', $value)
                    : $this->model->orWhere($field, '=', $value);
            }
        }
    }

    /**
     * 将给定的 whereIn 条件应用与模型
     * @param array $whereIn
     *
     * @return void
     * */
    protected function applyWhereInConditions(array $whereIn) : void
    {
        $field = array_key_first($whereIn);

        $this->model = $this->model->whereIn($field,$whereIn["$field"]);
    }
}

门面模式

使用 Laravel 框架必不可少的会用到它很多强大的门面类（Facades），门面提供了一个“静态”接口到服务容器中绑定的类。Laravel 内置了很多门面，你可能在不知道的情况下正在使用它们。

Laravel 的门面作为服务容器中的底层类的 “静态代理”，相比于传统静态方法，在维护时能够提供更加易于测试、更加灵活的、简明且富有表现力的语法。

门面就是一个提供访问容器中对象的类。该机制原理由 Facade 类实现，Laravel 自带的门面，以及创建的自定义门面，都会继承自 Illuminate\Support\Facades\Facade 基类。门面类只需要实现一个方法：getFacadeAccessor。正是 getFacadeAccessor 方法定义了从容器中解析什么，然后 Facade 基类使用魔术方法 __callStatic () 从你的门面中调用解析对象。

实现

在 app 目录下创建一个新的目录

  • Facades

在上面我们已经创建好了仓库模式，现在我们让仓库与门面合并使用，实现一个用户表的门面

<?php

namespace App\Facades;

use Illuminate\Support\Facades\Facade;

/**
 * @method static object with(array $relations)
 * @method static object lock(bool $lock = true)
 * @method static int count(array $data = [],bool $or = false,string $columns = '*')
 * @method static object create(array $data)
 * @method static void insert(array $data)
 * @method static mixed delete(array $data,bool $or = false)
 * @method static mixed filter(array $data,array $columns,bool $toArray = true)
 * @method static int update(array $data,array $columns,bool $or = false)
 * @method static void updateOrInsert(array $data,array $columns)
 * @method static int updateByWhereIn(array $data,array $in,array $update,bool $or = false)
 * @method static mixed find(int $id, array $columns = ['*'])
 * @method static mixed findByWhere(array $data, array $columns = ['*'], $or = false)
 * @method static mixed findByWhereIn(array $data,array $in,array $columns = ['*'],bool $or = false)
 * @method static object findByWhereAll(array $data, array $columns = ['*'], $or = false)
 * @method static object findByWhereInAll(array $data,array $in,array $columns = ['*'],bool $or = false)
 * */
class User extends Facade
{
    /**
     * 获取访问器
     *
     * @return string
     * */
    protected static function getFacadeAccessor() : string
    {
        return 'user';
    }

    /**
     * 获取类信息
     *
     * @return array
     * */
    public static function classFacadeInfo() : array
    {
        return [
            'class' => 'User',
            'facade_accessor' => 'user'
        ];
    }
}

创建一个服务提供者

<?php

namespace App\Providers;

use Illuminate\Support\ServiceProvider;

class FacadeServiceProvider extends ServiceProvider
{
    /**
     * Register services.
     *
     * @return void
     */
    public function register()
    {
        foreach (glob(app_path('Facades') . '/*') as $file)
        {
            $facade = '\App\Facades\\' . basename($file,'.php');

            $classFacadeInfo = $facade::classFacadeInfo();

            $repo = '\App\Repositories\Eloquent\\' . $classFacadeInfo['class'] . 'Repository';

            $this->app->singleton($classFacadeInfo['facade_accessor'], function () use ($repo) {
                return new $repo;
            });

            $this->app->alias($facade,$classFacadeInfo['class']);
        }
    }
}
